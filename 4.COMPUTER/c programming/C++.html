To find the even number from multiple inputed data.
                    
      
 #include <stdio.h>

 int main() {
   int n, count = 0;
   printf("Enter the number of elements in the list: ");
   scanf("%d", &n);

   int nums[n];
   printf("Enter %d numbers: ", n);
   for(int i=0; i<n; i++) {
     scanf("%d", &nums[i]);
     if(nums[i] % 2 == 0) {
      count++;
     }
   }

   printf("The number of even numbers in the list is %d.", count);
   return 0;
 }
Write a program to calculate the area of a circle given its radius.
                    
      
#include <stdio.h>

#define PI 3.14159

int main() {
    double radius, area;

    printf("Enter the radius of the circle: ");
    scanf("%lf", &radius);

    area = PI * radius * radius;

    printf("The area of the circle with radius %lf is %lf\n", radius, area);

    return 0;
}
